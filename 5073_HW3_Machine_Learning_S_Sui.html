# Create initiate empty list
within_cluster_variance = []

# build models for k from the range 2 to 9
for k in range(2, 9):
    kmeans = KMeans(n_clusters=k)
    kmeans.fit(dfr_scaled)
    within_cluster_variance.append(kmeans.inertia_)

#this could visualize the best cluster that we could find:
fig = plt.figure(figsize=(15, 5))
plt.plot(range(2, 9), within_cluster_variance, marker='o')
plt.grid(True)
plt.xlabel('Number of clusters')
plt.ylabel('Reduction in within cluster variation')
plt.title('Elbow curve');
